Create a cube in model space
Use camera transformation without translation (only direction)
Save position in model space to pass it to fragment shader

void Renderer::RenderSkybox(RenderFrameInfo& frameInfo)
{
	glDisable(GL_CULL_FACE);
	glUseProgram(shaderManager.GetProgram("Skybox"));
	// uniforms

	glm::mat4 mvp = frameInfo.Perspective * glm::mat4(glm::mat3(frameInfo.Camera));
	glUniformMatrix4fv(0, 1, GL_FALSE, &mvp[0][0]);

	// Texture
	glActiveTexture(GL_TEXTURE0);
	glBindTexture(GL_TEXTURE_CUBE_MAP, skybox);
	ShaderUtils::SetUniform(shaderManager.GetProgram("Skybox"), "textureData", 0);


	glBindVertexArray(skyboxVAO);
	glDrawArrays(GL_TRIANGLES, 0, skyBoxVertices.size());
	glEnable(GL_CULL_FACE);
}
